```
const PENDING = 'pending'
const FULFILLED = 'fulfilled'
const REJECTED = 'rejected'
function resolvePromise(p2, x, resolve, reject) {
  if (x === p2) {//返回相同promise
    throw new TypeError('same promise!')
  }
  if (x instanceof MyPromise) {//返回值为promise
    x.then(res => resolve(res), err => reject(err))
  } else {//返回值为普通值
    resolve(x)
  }
}
class MyPromise {
  state = PENDING
  result = undefined
  #handlers = []

  constructor(func) {
    console.log('开始构造函数！');
    console.log('定义resolve回调函数！');
    const resolve = (result) => {
      console.log('resolve回调函数调用！实参为' + result)
      if (this.state === PENDING) {//状态不可逆
        this.state = FULFILLED
        this.result = result
        this.#handlers.forEach(({ onFulfilled }) => {
          // console.log('onFulfilled回调函数调用！');
          onFulfilled(this.result)
        })
      }

    }
    console.log('定义reject回调函数！');
    const reject = (result) => {
      console.log('reject回调函数调用！实参为' + result)
      if (this.state === PENDING) {
        this.state = REJECTED
        this.result = result
        this.#handlers.forEach(({ onRejected }) => {
          //console.log('onFulfilled回调函数调用！');
          onRejected(this.result)
        })
      }
    }
    try {
      console.log('传入的回调函数调用！');
      func(resolve, reject)
    } catch (error) {
      reject(error)
    }

  }

  then(onFulfilled, onRejected) {
    console.log('then函数调用！');
    //判断传入的onFulfilled是不是函数
    onFulfilled = typeof onFulfilled === 'function' ? onFulfilled : x => x
    onRejected = typeof onRejected === 'function' ? onRejected : x => { throw x }
    //将返回值包装为promise对象
    const p2 = new MyPromise((resolve, reject) => {
      if (this.state === FULFILLED) {
        setTimeout(() => {//将所有的回调函数包装成异步函数
          try {
            console.log('onFulfilled回调函数调用！形参为' + this.result);
            const x = onFulfilled(this.result)
            resolvePromise(p2, x, resolve, reject)
          } catch (error) {
            reject(error)
          }
        }, 0);

      } else if (this.state === REJECTED) {
        setTimeout(() => {
          try {
            console.log('onRejected回调函数调用！形参为' + this.result);
            const x = onRejected(this.result)
            resolvePromise(p2, x, resolve, reject)
          } catch (error) {
            reject(error)
          }

        }, 0)

      } else {
        this.#handlers.push({
          onFulfilled: () => {
            setTimeout(() => {
              try {
                console.log('pending--onFulfilled回调函数调用！形参为' + this.result);
                const x = onFulfilled(this.result)
                resolvePromise(p2, x, resolve, reject)
              } catch (error) {
                reject(error)
              }

            }, 0)
          },
          onRejected: () => {
            setTimeout(() => {
              try {
                console.log('pending--onRejected回调函数调用！形参为' + this.result);
                const x = onRejected(this.result)
                resolvePromise(p2, x, resolve, reject)
              } catch (error) {
                reject(error)
              }
            }, 0)
          }
        })//如果resolve函数在异步函数中，则将回调函数注册到handlers数组中
      }
    })
    return p2
  }
  catch(onRejected) {
    return this.then(undefined, onRejected)
  }
  finally(onFinally) {
    return this.then(onFinally, onFinally)
  }

  static resolve(val) {
    // const p3 = new MyPromise((resolve) => {
    //   if (val instanceof MyPromise) {
    //     val.then(res => resolve(res))
    //   } else {
    //     resolve(val)
    //   }
    // })
    // return p3
    if (val instanceof MyPromise) {
      return val
    }
    return new MyPromise((resolve, reject) => {
      resolve(val)
    })

  }
  static rejected(val) {
    return new MyPromise((undefined, reject) => {
      reject(val)
    })
  }
  static race(args) {

    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(args)) {
        reject(new Error('not array!'))
      }
      args.forEach(p => {
        MyPromise.resolve(p).then(res => { resolve(res) }, err => { reject(err) })
      })
    })
  }
  static all(args) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(args)) {
        reject(new Error('not array'))
      }
      args.length === 0 && resolve(args)
      let count = 0
      const arr = []
      args.forEach((p, index) => {
        MyPromise.resolve(p).then(res => {
          arr[index] = res
          count++
          count === arr.length && resolve(arr)
        }, err => {
          reject(err)
        })
      })
    })
  }
  static allSettled(args) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(args)) {
        reject(new Error('not array'))
      }
      args.length === 0 && resolve(args)
      let count = 0
      const arr = []
      args.forEach((p, index) => {
        MyPromise.resolve(p).then(res => {
          arr[index] = { status: FULFILLED, value: res }
          count++
          count === arr.length && resolve(arr)
        }, err => {
          arr[index] = { status: REJECTED, value: err }
          count++
          count === arr.length && resolve(arr)
        })
      })
    })
  }
  static any(args) {
    return new MyPromise((resolve, reject) => {
      if (!Array.isArray(args)) {
        reject(new Error('not array'))
      }
      args.length === 0 && reject(new Error('all rejected'))
      let count = 0
      const errs = []
      args.forEach((p, index) => {
        MyPromise.resolve(p).then(res => {
          resolve(res)
        }, err => {
          errs[index] = err
          count++
          count === args.length && reject(new Error('all rejected'))
        })
      })
    })
  }

}
```

#### 2.手写call，bind函数

```
Function.prototype.myCall = function (obj, ...args) {
  obj.f = this
  const res = obj.f(...args)
  delete obj.f
  return res
}
Function.prototype.myApply = function (obj, args) {
  obj.f = this
  const res = obj.f(...args)
  delete obj.f
  return res
}
Function.prototype.myBind = function (obj, ...args) {
  return (...argsf) => {
    return this.call(obj, ...args, ...argsf)
  }
}
//-------------测试-------------
let obj = {
  age: 18
}
function func(a, b, c) {
  console.log(a + b + c)
  return this.age
}
console.log(func.myBind(obj, 1, 2)(3))
```

```
function sumMaker(length) {
  let nums = []
  function sum(...args) {
    nums.push(...args)
    if (nums.length < length) {
      return sum
    } else {
      return nums.slice(0, length).reduce((p, n) => p + n, 0)
    }
  }
  return sum
}
```

#### 4.防抖节流实现

```
function debounce(func, wait) {
  let timeId
  return function (...args) {
    let _this = this
    clearTimeout(timeId)
    timeId = setTimeout(() => {
      func.call(_this, ...args)
    }, wait);
  }
}
//-----------------测试----------
function func(val) {
  this.val = val
  console.log(this.val)
}
let test = debounce(func, 500)

let obj = { val: 1 }
obj.test = test
for (let i = 0; i < 10; i++) {
  obj.test(100)
}


function throttle(func, wait = 0) {
  let timeId
  return function (...args) {
    let _this = this
    if (timeId !== undefined) {//已经有定时器了，则直接返回
      return
    }
    timeId = setTimeout(() => {
      func.call(_this, ...args)
      timeId = undefined//本次定时器结束，将id设置为undefined
    }, wait);
  }
}
```

```
//浅拷贝
function copy(newObj, oldObj) {
  for (let k in oldObj) {
    if (oldObj.hasOwnProperty(k)) {
      newObj[k] = oldObj[k]
    }
  }
}
//深拷贝
function deepCopy(newObj, oldObj) {
  for (let k in oldObj) {
    if (oldObj.hasOwnProperty(k)) {
      if (oldObj[k] instanceof Array) {
        newObj[k] = []
        deepCopy(newObj[k], oldObj[k])
      } else if (oldObj[k] instanceof Object) {
        newObj[k] = {}
        deepCopy(newObj[k], oldObj[k])
      } else {
        newObj[k] = oldObj[k]
      }
    }
  }
}

}
```
